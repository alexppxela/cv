\def\@rubrichead#1{%
  \multicolumn{3}{@{}c}{%
    \@rubricfont%
%%    \makebox[0pt][l]{\colorbox[rgb]{0.31,0.42,0.73}{\makebox[\@almosttextwidth]{}}}%
    \makebox[0pt][l]{\colorbox[rgb]{0.51,0.62,0.83}{\makebox[\@almosttextwidth]{}}}%
    \makebox[\@almosttextwidth][c]{#1}\par\vspace{\rubricspace}}\\}
\renewcommand\subrubric[1]{%
  \@nextentry
  \gdef\@nextentry{}%
  &\multicolumn{2}{l}{%
    \raisebox{\@beforespace}{\@subrubricfont%
      \makebox[1ex][l]{\colorbox[gray]{0.7}{\textcolor[blue]{0.7}{M}}}#1}%
    \par\vspace{\subrubricspace}}\\}
\makeatother
